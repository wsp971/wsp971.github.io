1、简述下 slice与splice 的用法。

首先这两个方法数组中都有，其中slice 字符串中也有这个方法
slice 的定义是 arrayObject.slice(start,end)

slice 的作用是截取数组或者字符串，两个参数，start,表示开始下标，end代表结束下标。来看个例子:

var testString="abcdefghijklmn";
testString.slice(1)//bcdefghijklmn
testString.slice(1,5)//bcde
testString.slice(1,-1)//bcdefghijklm
// 数组
var testArray=[1,2,3,4,5];
testArray.slice(2)//3,4,5
testArray.slice(2,4)//3,4
testArray.slice(2,-2)//3

splice的定义是 arrayObject.splice(start,length,item1,item2,.....,itemn);

splice 的作用是删除数组中的部分数组，并且插入后面的item值到原数组中，返回删除的数组部分。来个例子：

var testArray=[1,2,3,4,5,6,7,8];
testArray.splice(2,3);//[3,4,5] 
console.log(testArray);//[1, 2, 6, 7, 8]
testArray.splice(1,3,"hello","world");//[2,3,4]
console.log(testArray)//[1,"hello","world",5,6,7,8]
testArray.splice(1)//[2,3,4,5,6,7,8]
console.log(testArray)//[1]

注意splice是分割数组，也就是这个操作会改变原数组，一定要记得

2.以下语句的执行输出结果是什么：

function fun(n,o){
    console.log(o);
    return {
        fun:function(m){
            fun(m,n);
        }
    }
}
var a=fun(0); a.fun(1);a.fun(2);a.fun(3);
var b=fun(0).fun(1).fun(2).fun(3);
var c=fun(0).fun(1);c.fun(2);c.fun(3);

这个题目主要考察的是闭包及作用域，其实前面的一篇博客已经讲的很清楚了，所以这个问题，我很轻松的就回答上了。

三行的输出是：

undefined 0 0 0 
undefined 0 1 2
undefined 0 1 1

简单的分析：

可以看到最外层的fun函数 打印了第二个参数值，并且返回了一个对象，这个对象里有一个函数，而这个函数的执行语句里引用了外层fun函数的变量n,即形成了一个闭包。
第一行就是考察，看你能否看懂这一点；第二行就是看你在循环执行的时候，变量o 的取值是否能够选择正确；第三行，也是同样的问题，只不过想混淆你。

3.有一个无序列表ul,里面有几个li，如何实现无缝滚动?

js如下：

 //整个定时滑动函数
 function moveAnimate(element){
        setInterval(function(){
            moveleft(element,500,500);
            element.appendChild(element.firstElementChild);
            element.style.marginLeft="0px";
        },3000);
}
//这个是滑动效果函数
function moveleft(element,length,time){
    var pie=length*100/time;
    var timer=setInterval(function(){
       if(length>0&&length>=pie){
           move(element,pie);
           length-=pie;
       }else{
           move(element,length);
         clearInterval(timer);
       }
    },100);
    function move(element,margin){
        var marginLeft=element.style.marginLeft.slice(0,-2);
        element.style.marginLeft=(Number(marginLeft)-margin)+"px";
    }
}
var ul=document.getElementsByTagName("ul")[0];
moveAnimate(ul);

这个例子是基于margin-left值的递减来实现动画的，当时我回答的是用position偏移来实现,面试官问可不可以有其他方式。我没有回答上
